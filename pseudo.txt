ALGORITMA Menghitung_take_home_pay_karyawan

begin 
    Deklarasi :
    jamKaryawan, gajiPokok, takeHomePay : integer
    gajiPokok = 1000000

    Implementasi :
    display "Selamat datang di program sederhana menghitung take home pay karyawan"

    display "Berapa jam karyawan bekerja dalam seminggu ini?"
    accept jamKaryawan

    if (jamKaryawan < 40) then
        takeHomePay = gajiPokok
    else 
        jamKaryawan = jamKaryawan - 40
        takeHomePay = gajiPokok + (jamKaryawan * 20000)
    endif

    display "Take home pay karyawan adalah Rp"
    display takeHomePay
end
